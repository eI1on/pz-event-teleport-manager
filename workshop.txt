title=[B41] Event Teleport Manager
description=This mod introduces a comprehensive event management system, allowing [b]server administrators[/b] to create scheduled events with specific times and locations while enabling [b]players to register and participate[/b].
description=
description=[h1]Core Features[/h1]
description=[b]Event Management System:[/b]
description=[list]
description=[*]Create events with custom names, descriptions, and precise timing
description=[*]Set exact coordinates for event locations
description=[*]Automatic event status tracking (Future > Active > Ended)
description=[*]Unique event IDs with 6-character codes for easy management
description=[/list]
description=
description=[b]Advanced Teleportation:[/b]
description=[list]
description=[*]Time-windowed teleportation (only during active events)
description=[*]Position protection system - saves original player locations
description=[*]Multiple return options: original position or safehouse
description=[*]Zombie density checks before teleporting (configurable threshold)
description=[*]Teleport cooldown system to prevent abuse
description=[/list]
description=
description=[b]Player Management:[/b]
description=[list]
description=[*]Flexible registration: admin-only or player self-registration
description=[*]Bulk player operations - manage multiple participants at once
description=[*]Real-time notifications when events start/end
description=[*]Permission-based access control
description=[/list]
description=
description=[h1]How to Use[/h1]
description=[b]For Players:[/b]
description=[list]
description=[*]Access via Client Panel (left side of screen)
description=[*]View available events and registration status
description=[*]Self-teleport during active event windows (if enabled, configurable in sandbox options)
description=[*]Return to original position or safehouse after events
description=[/list]
description=
description=[b]For Administrators:[/b]
description=[list]
description=[*]Full event management through Admin Panel or Client Panel
description=[*]Create, edit, and delete events with real-time updates
description=[*]Manage player registrations and teleport participants
description=[*]Monitor event status and player positions
description=[/list]
description=
description=[h1]Configuration Options[/h1]
description=Customize the mod behavior through sandbox settings:
description=[list]
description=[*][b]Event Admin Access:[/b] Set minimum permission level (None to Admin)
description=[*][b]Player Registration:[/b] Enable/disable self-registration
description=[*][b]Teleport Return Mode:[/b] Choose return behavior (Original, Safehouse, or Both)
description=[*][b]Zombie Check:[/b] Enable safety checks before teleporting
description=[*][b]Zombie Threshold:[/b] Maximum zombie count for safe teleportation (0-50)
description=[*][b]Teleport Cooldown:[/b] Delay between teleports (0-3600 seconds)
description=[*][b]Position Protection:[/b] Automatically save player positions
description=[/list]
description=
description=[h1]Safety & Security Features[/h1]
description=[list]
description=[*]Automatic position saving before teleportation
description=[*]Zombie density validation to prevent dangerous teleports
description=[*]Permission-based access control system
description=[*]Teleport cooldown to prevent spam/abuse
description=[*]Event status validation before allowing actions
description=[*]Comprehensive logging for admin oversight
description=[/list]
description=
description=[h1]User Interface[/h1]
description=[list]
description=[*]Intuitive dual-interface design for admins and players
description=[*]Real-time event status updates
description=[*]Clear registration and teleportation feedback
description=[*]Responsive notifications and status messages
description=[*]Easy bulk operations for efficient management
description=[/list]

description=[*]Automatic updates pushed to all connected players
